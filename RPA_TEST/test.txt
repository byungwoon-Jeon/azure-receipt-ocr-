def query_rpa_ccr_line_summ(duser_input: dict, db_config: dict, from_date: str, to_date: str) -> list:
    """
    RPA_CCR_LINE_SUMM에서 CREATE_DATE 기준으로 데이터 조회
    - 키값(FIID, LINE_INDEX, RECEIPT_INDEX, COMMON_YN) + CONTENTS + RESULT_CODE 반환
    """
    query_sql = """
        SELECT 
            FIID,
            LINE_INDEX,
            RECEIPT_INDEX,
            COMMON_YN,
            CONTENTS,
            RESULT_CODE
        FROM RPA_CCR_LINE_SUMM
        WHERE CREATE_DATE >= :from_date
          AND CREATE_DATE <  :to_date
        ORDER BY CREATE_DATE, FIID, LINE_INDEX, RECEIPT_INDEX
    """

    params = {"from_date": from_date, "to_date": to_date}

    return idp_db_manager.excute_queris(
        quries=[query_sql],
        params=[params],
        db_config=db_config
    ) or []



import re
import logging
from typing import Optional

logger = logging.getLogger("APPROVAL_NO")

def extract_approval_no(contents: str) -> Optional[str]:
    """
    CONTENTS 문자열에서 승인번호(승인번호/승인NO/승인코드/Auth No/Auth Code/Approval No 등)를 추출한다.
    - 라벨과 값이 같은 줄 or 다른 줄(바로 다음 줄) 모두 처리
    - 값은 6~10자리 숫자 또는 영문+숫자 조합만 인정
    - 여러 개 발견 시 첫 번째만 반환, 나머지는 로그로 남김
    - '1234 5678' 같은 공백 포함 케이스도 지원 (→ 12345678로 변환)
    """
    if not contents:
        return None

    # 라벨 패턴
    label_pattern = re.compile(
        r"(승인\s*(번호|NO|코드)|Auth(entication)?(\s*No|\s*Code)?|Approval\s*No)",
        re.IGNORECASE
    )

    # 값 패턴 (연속 6~10자리 or 4+4 공백 포함)
    value_pattern = re.compile(r"[0-9A-Za-z]{6,10}|\d{4}\s+\d{4}")

    lines = contents.splitlines()
    candidates = []

    for i, line in enumerate(lines):
        if label_pattern.search(line):
            # 1) 같은 줄에서 값 찾기
            m = value_pattern.search(line)
            if m:
                candidates.append(m.group().replace(" ", ""))
                continue

            # 2) 다음 줄에서 값 찾기
            if i + 1 < len(lines):
                m = value_pattern.search(lines[i + 1])
                if m:
                    candidates.append(m.group().replace(" ", ""))

    if not candidates:
        return None

    if len(candidates) > 1:
        logger.warning("[승인번호] 여러 개 발견됨 → %s (첫 번째만 사용)", candidates)

    return candidates[0]


def update_approval_no(db_config: dict, record: dict, approval_no: str) -> None:
    """
    RPA_CCR_LINE_SUMM 테이블에서 특정 키(FIID, LINE_INDEX, RECEIPT_INDEX, COMMON_YN)를 기준으로
    APPROVAL_NO와 UPDATE_DATE를 무조건 갱신한다.
    업데이트 성공 시 영향받은 행(rowcount)을 로그로 남긴다.
    """
    query_sql = """
        UPDATE MISADM.RPA_CCR_LINE_SUMM
        SET APPROVAL_NO = :approval_no,
            UPDATE_DATE = CURRENT_TIMESTAMP
        WHERE FIID = :fiid
          AND LINE_INDEX = :line_index
          AND RECEIPT_INDEX = :receipt_index
          AND COMMON_YN = :common_yn
    """

    params = {
        "approval_no": approval_no,
        "fiid": record.get("FIID"),
        "line_index": record.get("LINE_INDEX"),
        "receipt_index": record.get("RECEIPT_INDEX"),
        "common_yn": record.get("COMMON_YN"),
    }

    try:
        result = idp_db_manager.excute_queris(
            quries=[query_sql],
            params=[params],
            db_config=db_config
        )
        # HANA에서는 보통 영향받은 행 수가 반환됨 → 없으면 result 자체 출력
        rowcount = result[0].get("ROWCOUNT") if result and isinstance(result, list) else None

        logging.info(
            f"[UPDATE] 승인번호 갱신 (강제 덮어쓰기) → "
            f"FIID={params['fiid']} LINE={params['line_index']} RECEIPT={params['receipt_index']} "
            f"APPROVAL_NO={approval_no} | rowcount={rowcount}"
        )
    except Exception as e:
        logging.error(f"[ERROR] 승인번호 갱신 실패 → {params} | {e}")
