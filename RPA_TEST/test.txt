def estimate_elapsed_time(request_data: dict) -> dict:
    """
    idp_items 개수에 따라 예상 소요 시간을 계산합니다.
    - 아이템 1개당 15초 기준

    Parameters
    ----------
    request_data : dict
        {
            "systemName": "ABC",
            "idp_items": [
                {"ppeId": "x", "ppeSeq": "1", "url": "..."}, ...
            ]
        }

    Returns
    -------
    dict
        {"elp_time": <예상 소요 시간(초)>}
    """
    try:
        # idp_items가 존재하는지 확인
        items = request_data.get("idp_items", [])
        if not isinstance(items, list):
            raise ValueError("idp_items must be a list")

        item_count = len(items)
        time_per_item = 15  # 초 단위 (기준 처리 시간)
        elp_time = item_count * time_per_item

        response_data = {"elp_time": elp_time}
        return response_data

    except Exception as e:
        # 예상치 못한 오류 시 안전하게 처리
        return {"error": f"Failed to calculate elapsed time: {e}"}
