import os

def check_file_size_under_15mb(file_path, limit_mb=15):
    """
    파일 경로를 입력받아 파일 크기를 확인하고,
    15MB 이하인지 판별한다.

    Returns:
        size_mb (float), is_under_limit (bool)
    """
    if not os.path.isfile(file_path):
        raise FileNotFoundError(f"File not found: {file_path}")

    size_bytes = os.path.getsize(file_path)
    size_mb = size_bytes / (1024 * 1024)

    is_under_limit = size_mb <= limit_mb
    return size_mb, is_under_limit


# 사용 예시
if __name__ == "__main__":
    path = r"C:\example\test.pdf"
    size_mb, ok = check_file_size_under_15mb(path)

    print(f"File size: {size_mb:.2f} MB")
    if ok:
        print("✔ File size is within the 15MB limit.")
    else:
        print("✘ File exceeds the 15MB limit.")
