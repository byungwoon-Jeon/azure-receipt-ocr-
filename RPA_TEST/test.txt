# 특정 문장이 텍스트 파일에서 몇 번 등장하는지 카운트하는 코드

def count_phrase_in_file(file_path: str, phrase: str) -> int:
    try:
        with open(file_path, 'r', encoding='utf-8') as f:
            text = f.read()
        return text.count(phrase)
    except Exception as e:
        print(f"Error: {e}")
        return 0

# 사용 예시
file_path = "sample.txt"
phrase = "찾을문장"

count = count_phrase_in_file(file_path, phrase)
print(f"'{phrase}' 은(는) {count}번 등장합니다.")
